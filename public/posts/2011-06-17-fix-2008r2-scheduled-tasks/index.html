Function Change-ScheduledTask {
Param($XMLIn)
$ShortDate = [regex]"(?<shortdate>[0-9]{4}/.-/.-T(?:2[0-3]|[01][0-9])[:.][0-5][0-9][:.][0-5][0-9])(?<digits>.\d*)"

foreach ($line in $XMLIn){
If ($line -match “$ShortDate”) {
$line = [regex]::Replace($line, $ShortDate, $($Matches[“Shortdate”]))
}
$line

}
}
